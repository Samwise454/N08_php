CREATE TABLE users(
	id INT(10) PRIMARY KEY NOT NULL AUTO_INCREMENT,
    firstname VARCHAR(150) NOT NULL,
    lastname VARCHAR(150) NOT NULL,
    nickname VARCHAR(150) NOT NULL,
    email VARCHAR(200) NOT NULL,
    tel VARCHAR(20) NOT NULL,
    lastclass VARCHAR(8) NOT NULL,
    house VARCHAR(8) NOT NULL,
    password VARCHAR(200) NOT NULL,
    img VARCHAR(100) NOT NULL,
    token VARCHAR(150) NOT NULL,
    date_created TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE bizdata(
    id INT(10) NOT NULL PRIMARY KEY AUTO_INCREMENT,
    bizname VARCHAR(150) NOT NULL,
    bizdesc VARCHAR(500) NOT NULL,
    bizlogo VARCHAR(100) NOT NULL,
    bizimg1 VARCHAR(100) NOT NULL,
    bizimg2 VARCHAR(100) NOT NULL,
    social VARCHAR(150) NOT NULL,
    date_created TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
);
